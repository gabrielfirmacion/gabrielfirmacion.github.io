Integers

times

5.times do |x|
print x
end
#=> 0 1 2 3 4

even?

x = any integer
x.even? #=> returns true if it is, otherwise returns false

odd?

x = any integer
x.odd? #=> returns true if it is, otherwise returns false

next

1.next
#=> 2

String

include?

"bootcamp".include? "mp" #=> true

length

"bootcamp".length #=> 8, returns the number of characters in string

delete

"bootcamp".delete "o" #=> "btcamp"

reverse

"bootcamp".reverse #=> "pmactoob"

Array

map!

a = ['g', 'a', 'b', 'r', 'i', 'e', 'l']
a.map! {|x| x + "!"}
a #=> ['g!', 'a!', 'b!', 'r!', 'i!', 'e!', 'l!']

sort!

a = ['x', 'a', 'n', 'd', 'e', 'r']
a.sort! #=> ['a', 'd', 'e', 'n', 'r', 'x']
a.sort! {|x,y| y <=> x} #=> ['x', 'r', 'n', 'e', 'd', 'a']

push

a = ['a', 'b', 'c', 'd']
a.push('e', 1, 2, true)
#=> ['a', 'b', 'c', 'd', 'e', true, 2, ]

pop

a = ['a', 'b', 'c', 'd', 'e', true, 2]
a.pop #=> 2
#=> ['a', 'b', 'c', 'd', 'e', true]
a.pop(3) #=> ['d', 'e', true]
#=> ['a', 'b', 'c']

Hash

has_key?

h = {"a" = 10, "b" = 20}
h.has_key?("b") #=> true
h.has_key?("c") #=> false

has_value?

h = {"a" = 10, "b" = 20}
h.has_value?(20) #=> true
h.has_value?(77) #=> false

each_key

h = {"a" = 10, "b" = 20}
h.each_key {|key| puts key}
#=> a
#=> b

invert

h = { "a" = 10, "b" = 20, "c" = 30, "d" = 40 }
h.invert
#=> { 10 = "a", 20 = "b", 30 = "c", 40 = "d"}
